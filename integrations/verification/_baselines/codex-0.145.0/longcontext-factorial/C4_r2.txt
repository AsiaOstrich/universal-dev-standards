const SENSITIVE_PATTERN = /(api[_-]?key|token|password|secret|bearer)/i;

export function redactSensitiveFields(obj: Record<string, unknown>): Record<string, unknown> {
  const result: Record<string, unknown> = {};
  for (const [key, value] of Object.entries(obj)) {
    if (SENSITIVE_PATTERN.test(key)) {
      result[key] = "[REDACTED]";
    } else if (Array.isArray(value)) {
      result[key] = value.map((item) =>
        item !== null && typeof item === "object"
          ? redactSensitiveFields(item as Record<string, unknown>)
          : item
      );
    } else if (value !== null && typeof value === "object") {
      result[key] = redactSensitiveFields(value as Record<string, unknown>);
    } else {
      result[key] = value;
    }
  }
  return result;
}
export async function computeKeyHash(key: string): Promise<string> {
  const encoder = new TextEncoder();
  const data = encoder.encode(key);
  const hashBuffer = await crypto.subtle.digest("SHA-256", data);
  const hashArray = Array.from(new Uint8Array(hashBuffer));
  return hashArray.map((b) => b.toString(16).padStart(2, "0")).join("");
}

export async function validateApiKey(
  key: string | null,
  db: D1Database,
  publicKey?: string
): Promise<boolean> {
  if (!key) return false;
  if (publicKey && key === publicKey) return true;

  const hash = await computeKeyHash(key);

  const row = await db
    .prepare("SELECT id FROM api_keys WHERE key_hash = ? AND is_active = 1")
    .bind(hash)
    .first<{ id: string }>();

  if (!row) return false;

  await db
    .prepare("UPDATE api_keys SET last_used_at = datetime('now') WHERE id = ?")
    .bind(row.id)
    .run();

  return true;
}

/**
 * Q1：各 Arm 總 token cost（USD）
 *
 * 對齊 XSPEC-191 §5.4 Q1。
 * 使用 event_type='performance' + event_subtype='token_usage'，
 * 從 event_data JSON 提取 cost_usd 與 cohort.arm。
 *
 * 回傳欄位：arm / total_cost_usd / session_count / total_input_tokens / total_output_tokens
 *
 * @param arm - 指定 Arm（'a' | 'b' | 'c'），若為 undefined 則查詢所有 Arm
 */
export function buildQ1ArmTokenCost(arm?: 'a' | 'b' | 'c'): string {
  const armFilter = arm
    ? `AND JSON_EXTRACT(event_data, '$.cohort.arm') = '${arm}'`
    : '';

  return `
SELECT
  JSON_EXTRACT(event_data, '$.cohort.arm')           AS arm,
  COUNT(DISTINCT session_id)                          AS session_count,
  SUM(CAST(JSON_EXTRACT(event_data, '$.input_tokens')  AS INTEGER)) AS total_input_tokens,
  SUM(CAST(JSON_EXTRACT(event_data, '$.output_tokens') AS INTEGER)) AS total_output_tokens,
  ROUND(SUM(CAST(JSON_EXTRACT(event_data, '$.cost_usd') AS REAL)), 4) AS total_cost_usd,
  ROUND(AVG(CAST(JSON_EXTRACT(event_data, '$.cost_usd') AS REAL)), 4) AS avg_cost_per_session_usd
FROM events_v2
WHERE event_type    = 'performance'
  AND event_subtype = 'token_usage'
  AND client_id     LIKE 'PoC-%'
  AND JSON_EXTRACT(event_data, '$.cohort.arm') IS NOT NULL
  ${armFilter}
GROUP BY arm
ORDER BY arm;
`.trim();
}

/**
 * Q2：各 Arm 各 stage 成本佔比
 *
 * 對齊 XSPEC-191 §5.4 Q2。
 * 每個 stage 的 token_usage cost_usd 加總，計算佔 Arm 總成本的百分比。
 *
 * 回傳欄位：arm / stage / stage_cost_usd / cost_pct
 *
 * @param arm - 指定 Arm，若為 undefined 則查詢所有 Arm
 */
export function buildQ2StageCostBreakdown(arm?: 'a' | 'b' | 'c'): string {
  const armFilter = arm
    ? `AND JSON_EXTRACT(event_data, '$.cohort.arm') = '${arm}'`
    : '';

  return `
WITH arm_total AS (
  SELECT
    JSON_EXTRACT(event_data, '$.cohort.arm') AS arm,
    SUM(CAST(JSON_EXTRACT(event_data, '$.cost_usd') AS REAL)) AS arm_total_cost
  FROM events_v2
  WHERE event_type    = 'performance'
    AND event_subtype = 'token_usage'
    AND client_id     LIKE 'PoC-%'
    AND JSON_EXTRACT(event_data, '$.cohort.arm') IS NOT NULL
    ${armFilter}
  GROUP BY arm
)
SELECT
  e.arm,
  e.stage,
  ROUND(SUM(e.cost_usd), 4)                                       AS stage_cost_usd,
  ROUND(SUM(e.cost_usd) * 100.0 / MAX(t.arm_total_cost), 2)      AS cost_pct
FROM (
  SELECT
    JSON_EXTRACT(event_data, '$.cohort.arm') AS arm,
    stage,
    CAST(JSON_EXTRACT(event_data, '$.cost_usd') AS REAL)         AS cost_usd
  FROM events_v2
  WHERE event_type    = 'performance'
    AND event_subtype = 'token_usage'
    AND client_id     LIKE 'PoC-%'
    AND JSON_EXTRACT(event_data, '$.cohort.arm') IS NOT NULL
    ${armFilter}
) e
JOIN arm_total t ON t.arm = e.arm
GROUP BY e.arm, e.stage
ORDER BY e.arm, cost_pct DESC;
`.trim();
}

/**
 * Q3：各 Arm Cache Hit Ratio 對比
 *
 * 對齊 XSPEC-191 §5.4 Q3。
 * 使用 event_type='performance' + event_subtype='cache_metric'，
 * 從 event_data JSON 提取 cache_read_tokens / prompt_tokens（對應 TokenUsageData.input_tokens）。
 * cache_hit_ratio 直接取 CacheMetricData.cache_hit_ratio 欄位。
 *
 * 回傳欄位：arm / avg_cache_hit_ratio / total_cache_read_tokens / total_prompt_tokens
 */
export function buildQ3CacheHitRatio(arm?: 'a' | 'b' | 'c'): string {
  const armFilter = arm
    ? `AND JSON_EXTRACT(event_data, '$.cohort.arm') = '${arm}'`
    : '';

  return `
SELECT
  JSON_EXTRACT(event_data, '$.cohort.arm')  AS arm,
  SUM(CAST(JSON_EXTRACT(event_data, '$.cache_read_tokens') AS INTEGER))
                                             AS total_cache_read_tokens,
  SUM(CAST(JSON_EXTRACT(event_data, '$.cache_write_tokens') AS INTEGER))
                                             AS total_cache_write_tokens,
  ROUND(
    AVG(CAST(JSON_EXTRACT(event_data, '$.cache_hit_ratio') AS REAL)),
    4
  )                                          AS avg_cache_hit_ratio,
  ROUND(
    SUM(CAST(JSON_EXTRACT(event_data, '$.cache_read_tokens') AS REAL)) * 100.0
    / NULLIF(
        SUM(CAST(JSON_EXTRACT(event_data, '$.cache_read_tokens') AS REAL))
        + SUM(CAST(JSON_EXTRACT(event_data, '$.cache_write_tokens') AS REAL)),
        0
      ),
    2
  )                                          AS cache_hit_ratio_pct
FROM events_v2
WHERE event_type    = 'performance'
  AND event_subtype = 'cache_metric'
  AND client_id     LIKE 'PoC-%'
  AND JSON_EXTRACT(event_data, '$.cohort.arm') IS NOT NULL
  ${armFilter}
GROUP BY arm
ORDER BY arm;
`.trim();
}

  if (present.length < 2) return;

  const where = present.map((f) => `${f} = ?`).join(" OR ");
  const binds = present.map((f) => ids[f] as string);
  const existing = await db
    .prepare(
      `SELECT subject_id, user_id, installation_id, client_id FROM identity_map WHERE ${where} LIMIT 1`
    )
    .bind(...binds)
    .first<{
      subject_id: string;
      user_id: string | null;
      installation_id: string | null;
      client_id: string | null;
    }>();

  if (existing) {
    await db
      .prepare(
        `UPDATE identity_map SET user_id = ?, installation_id = ?, client_id = ?, updated_at = datetime('now') WHERE subject_id = ?`
      )
      .bind(
        existing.user_id ?? ids.user_id ?? null,
        existing.installation_id ?? ids.installation_id ?? null,
        existing.client_id ?? ids.client_id ?? null,
        existing.subject_id
      )
      .run();
  } else {
    await db
      .prepare(
        `INSERT INTO identity_map (subject_id, user_id, installation_id, client_id) VALUES (?, ?, ?, ?)`
      )
      .bind(
        crypto.randomUUID(),
        ids.user_id ?? null,
        ids.installation_id ?? null,
        ids.client_id ?? null
      )
      .run();
  }
}

/** Expand one identifier to all known identifiers of the same subject. */
export async function expandSubject(
  db: D1Database,
  type: IdField,
  value: string
): Promise<SubjectIds> {
  const row = await db
    .prepare(
      `SELECT user_id, installation_id, client_id FROM identity_map WHERE ${type} = ? LIMIT 1`
    )
    .bind(value)
    .first<{
      user_id: string | null;
      installation_id: string | null;
      client_id: string | null;
    }>();

  if (!row) {
    // Degrade: no linkage — erase only what directly matches this identifier.
    return { [type]: value } as SubjectIds;
  }
  return {
    user_id: row.user_id ?? undefined,
    installation_id: row.installation_id ?? undefined,
    client_id: row.client_id ?? undefined,
  };
}

export interface ErasureResult {
  status: 200 | 400;
  body: {
    erased?: boolean;
    expanded_identifiers?: IdField[];
    affected_counts?: Record<string, number>;
    error?: string;
  };
}

/**
 * Execute erasure: expand subject → hard-delete across matching tables →
 * write audit (hash only, no raw PII) → return per-table counts.
 * Aggregation tables (aggregations_weekly/monthly) are anonymised and excluded
 * (GDPR aggregate exemption; declared in Privacy Policy).
 */
export async function eraseSubject(
  db: D1Database,
  type: string,
  value: string,
  actor: string
): Promise<ErasureResult> {
  if (!ID_FIELDS.includes(type as IdField)) {
    return { status: 400, body: { error: `Invalid identifier_type: ${type}` } };
  }
  if (!value) {
    return { status: 400, body: { error: "Missing identifier_value" } };
  }

  const subject = await expandSubject(db, type as IdField, value);
  const counts: Record<string, number> = {};

  for (const field of ID_FIELDS) {
    const v = subject[field];
    if (!v) continue;
    for (const table of TABLES_BY_FIELD[field]) {
      const res = await db
        .prepare(`DELETE FROM ${table} WHERE ${field} = ?`)
        .bind(v)
        .run();
      counts[table] = (counts[table] ?? 0) + (res.meta?.changes ?? 0);
    }
  }

  // Remove the identity_map row(s) for this subject too.
  for (const field of ID_FIELDS) {
    const v = subject[field];
    if (v) {
      await db.prepare(`DELETE FROM identity_map WHERE ${field} = ?`).bind(v).run();
    }
  }

  const resolved = ID_FIELDS.filter((f) => subject[f]);
  const idHash = await computeKeyHash(value);
  await db
    .prepare(
      `INSERT INTO erasure_audit (id, identifier_type, identifier_hash, expanded_ids, affected_counts, actor)
       VALUES (?, ?, ?, ?, ?, ?)`
    )
    .bind(
      crypto.randomUUID(),
      type,
      idHash,
      JSON.stringify(resolved),
      JSON.stringify(counts),
      actor
    )
    .run();

  return {
    status: 200,
    body: { erased: true, expanded_identifiers: resolved, affected_counts: counts },
  };
}

const COHORT_KEYS = [
  "edition:community",
  "edition:commercial",
  "edition:trial",
] as const;

type CohortKey = (typeof COHORT_KEYS)[number];

interface WeeklyMetric {
  metric_name: string;
  metric_value: number | null;
  metric_data: string | null;
  sample_size: number;
}

async function computeMetricsForCohort(
  db: D1Database,
  cohortKey: CohortKey,
  weekStart: string,
): Promise<WeeklyMetric[]> {
  const edition = cohortKey.split(":")[1];
  const weekEnd = new Date(weekStart);
  weekEnd.setDate(weekEnd.getDate() + 7);
  const weekEndStr = weekEnd.toISOString().slice(0, 10);

  const baseWhere = edition === "all"
    ? `WHERE timestamp >= ? AND timestamp < ?`
    : `WHERE edition = ? AND timestamp >= ? AND timestamp < ?`;
  const baseBinds: unknown[] = edition === "all"
    ? [weekStart, weekEndStr]
    : [edition, weekStart, weekEndStr];

  const metrics: WeeklyMetric[] = [];

  // Gate pass rate（quality events）
  const gateResult = await db
    .prepare(
      `SELECT
         COUNT(*) AS total,
         SUM(CASE WHEN json_extract(event_data, '$.passed') = 1 THEN 1 ELSE 0 END) AS passed
       FROM events_v2
       ${baseWhere}
         AND event_type = 'quality'
         AND event_subtype IN ('gate_pass', 'gate_fail')`
    )
    .bind(...baseBinds)
    .first<{ total: number; passed: number }>();

  metrics.push({
    metric_name: "gate_pass_rate",
    metric_value: gateResult && gateResult.total > 0
      ? Math.round((gateResult.passed / gateResult.total) * 10000) / 10000
      : null,
    metric_data: null,
    sample_size: gateResult?.total ?? 0,
  });

  // License compliance block rate（quality events）
  const licenseResult = await db
    .prepare(
      `SELECT
         COUNT(*) AS total,
         SUM(CASE WHEN json_extract(event_data, '$.blocked_count') > 0 THEN 1 ELSE 0 END) AS blocked
       FROM events_v2
       ${baseWhere}
         AND event_type = 'quality'
         AND event_subtype = 'license_compliance_result'`
    )
    .bind(...baseBinds)
    .first<{ total: number; blocked: number }>();

  metrics.push({
    metric_name: "license_compliance_block_rate",
    metric_value: licenseResult && licenseResult.total > 0
      ? Math.round((licenseResult.blocked / licenseResult.total) * 10000) / 10000
      : null,
    metric_data: null,
    sample_size: licenseResult?.total ?? 0,
  });

  // Ship decision distribution（outcome events）
  const shipResult = await db
    .prepare(
      `SELECT
         json_extract(event_data, '$.decision') AS decision,
         COUNT(*) AS cnt
       FROM events_v2
       ${baseWhere}
         AND event_type = 'outcome'
         AND event_subtype = 'ship_decision'
       GROUP BY json_extract(event_data, '$.decision')`
    )
    .bind(...baseBinds)
    .all<{ decision: string; cnt: number }>();

  const shipDist: Record<string, number> = {};
  let shipTotal = 0;
  for (const row of shipResult.results ?? []) {
    shipDist[row.decision] = row.cnt;
    shipTotal += row.cnt;
  }

  metrics.push({
    metric_name: "ship_decision_distribution",
    metric_value: null,
    metric_data: JSON.stringify(shipDist),
    sample_size: shipTotal,
  });

  // AI proposal acceptance rate（decision events）
  const proposalResult = await db
    .prepare(
      `SELECT
         COUNT(*) AS total,
         SUM(CASE WHEN json_extract(event_data, '$.outcome') = 'accepted' THEN 1 ELSE 0 END) AS accepted
       FROM events_v2
       ${baseWhere}
         AND event_type = 'decision'
         AND event_subtype = 'ai_proposal_outcome'`
    )
    .bind(...baseBinds)
    .first<{ total: number; accepted: number }>();

  metrics.push({
    metric_name: "ai_proposal_acceptance_rate",
    metric_value: proposalResult && proposalResult.total > 0
      ? Math.round((proposalResult.accepted / proposalResult.total) * 10000) / 10000
      : null,
    metric_data: null,
    sample_size: proposalResult?.total ?? 0,
  });

  return metrics;
}

export async function computeWeeklyAggregations(db: D1Database): Promise<void> {
  // 計算上一個完整周（周一為起點）
  const now = new Date();
  const dayOfWeek = now.getUTCDay(); // 0=Sun, 1=Mon, ...
  const daysToLastMonday = dayOfWeek === 0 ? 6 : dayOfWeek - 1;
  const lastMonday = new Date(now);
  lastMonday.setUTCDate(now.getUTCDate() - daysToLastMonday - 7);
  const weekStart = lastMonday.toISOString().slice(0, 10);

  for (const cohortKey of COHORT_KEYS) {
    const metrics = await computeMetricsForCohort(db, cohortKey, weekStart);
    for (const m of metrics) {
      await db
        .prepare(
          `INSERT OR REPLACE INTO aggregations_weekly
             (week_start, cohort_key, metric_name, metric_value, metric_data, sample_size)
           VALUES (?, ?, ?, ?, ?, ?)`
        )
        .bind(weekStart, cohortKey, m.metric_name, m.metric_value, m.metric_data, m.sample_size)
        .run();
    }
  }
}

export async function computeMonthlyAggregations(db: D1Database): Promise<void> {
  // 計算上一個完整月
  const now = new Date();
  const firstOfThisMonth = new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth(), 1));
  const firstOfLastMonth = new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth() - 1, 1));
  const monthStart = firstOfLastMonth.toISOString().slice(0, 10);
  const monthEnd = firstOfThisMonth.toISOString().slice(0, 10);

  // 月聚合：ship decision + abort distribution（全量，不分 cohort）
  const shipResult = await db
    .prepare(
      `SELECT
         json_extract(event_data, '$.decision') AS decision,
         COUNT(*) AS cnt
       FROM events_v2
       WHERE timestamp >= ? AND timestamp < ?
         AND event_type = 'outcome'
         AND event_subtype = 'ship_decision'
       GROUP BY json_extract(event_data, '$.decision')`
    )
    .bind(monthStart, monthEnd)
    .all<{ decision: string; cnt: number }>();

  const shipDist: Record<string, number> = {};
  let shipTotal = 0;
  for (const row of shipResult.results ?? []) {
    shipDist[row.decision] = row.cnt;
    shipTotal += row.cnt;
  }

  await db
    .prepare(
      `INSERT OR REPLACE INTO aggregations_monthly
         (month_start, cohort_key, metric_name, metric_value, metric_data, sample_size)
       VALUES (?, ?, ?, ?, ?, ?)`
    )
    .bind(monthStart, "global", "ship_decision_distribution", null, JSON.stringify(shipDist), shipTotal)
    .run();

  // Abort trigger distribution
  const abortResult = await db
    .prepare(
      `SELECT
         json_extract(event_data, '$.trigger') AS trigger,
         COUNT(*) AS cnt
       FROM events_v2
       WHERE timestamp >= ? AND timestamp < ?
         AND event_type = 'outcome'
         AND event_subtype = 'abort_triggered'
       GROUP BY json_extract(event_data, '$.trigger')`
    )
    .bind(monthStart, monthEnd)
    .all<{ trigger: string; cnt: number }>();

  const abortDist: Record<string, number> = {};
  let abortTotal = 0;
  for (const row of abortResult.results ?? []) {
    abortDist[row.trigger] = row.cnt;
    abortTotal += row.cnt;
  }

  await db
    .prepare(
      `INSERT OR REPLACE INTO aggregations_monthly
         (month_start, cohort_key, metric_name, metric_value, metric_data, sample_size)
       VALUES (?, ?, ?, ?, ?, ?)`
    )
    .bind(monthStart, "global", "abort_trigger_distribution", null, JSON.stringify(abortDist), abortTotal)
    .run();
}

## Purpose

This skill provides systematic checklists for code review and pre-commit verification.

## Quick Reference

### Comment Prefixes

| Prefix | Meaning | Action Required |
|--------|---------|------------------|
| **❗ BLOCKING** | Must fix before merge | 🔴 Required |
| **⚠️ IMPORTANT** | Should fix, but not blocking | 🟡 Recommended |
| **💡 SUGGESTION** | Nice-to-have improvement | 🟢 Optional |
| **❓ QUESTION** | Need clarification | 🔵 Discuss |
| **📝 NOTE** | Informational, no action | ⚪ Informational |

### Review Checklist Categories

1. **Functionality** - Does it work?
2. **Design** - Right architecture?
3. **Quality** - Clean code?
4. **Readability** - Easy to understand?
5. **Tests** - Adequate coverage?
6. **Security** - No vulnerabilities?
7. **Performance** - Efficient?
8. **Errors** - Properly handled?
9. **Docs** - Updated?
10. **Dependencies** - Necessary?

### Pre-Commit Checklist

- [ ] Build succeeds (zero errors, zero warnings)
- [ ] All tests pass
- [ ] Code follows project standards
- [ ] No security vulnerabilities
- [ ] Documentation updated
- [ ] Branch synced with target

## Detailed Guidelines

For complete standards, see:
- [Review Checklist](./review-checklist.md)
- [Pre-Commit Checklist](./checkin-checklist.md)

### AI-Optimized Format (Token-Efficient)

For AI assistants, use the YAML format file for reduced token usage:
- Base standard: `ai/standards/code-review.ai.yaml`

## Example Review Comments

```markdown
❗ BLOCKING: Potential SQL injection vulnerability here.
Please use parameterized queries instead of string concatenation.

⚠️ IMPORTANT: This method is doing too much (120 lines).
Consider extracting validation logic to a separate method.

💡 SUGGESTION: Consider using a Map here instead of an array for O(1) lookup.

❓ QUESTION: Why are we using setTimeout here instead of async/await?

📝 NOTE: This is a clever solution! Nice use of reduce here.
```

## Core Principles

1. **Be Respectful** - Review code, not the person
2. **Be Thorough** - Check functionality, not just syntax
3. **Be Timely** - Review within 24 hours
4. **Be Clear** - Explain WHY, not just WHAT

  - standards: "naming, formatting, comments"
  - no_smells: "methods≤50 lines, nesting≤3, complexity≤10, no duplication"
  - security: "no hardcoded secrets, no SQLi, no XSS, no insecure deps"
  verify: "run linter, static analysis, security scanner"

docs:
  - api_docs: "public APIs documented"
  - readme: "updated if needed"
  - changelog: "user-facing changes → [Unreleased]"

workflow:
  - branch_naming: "feature/, fix/, docs/, chore/"
  - commit_message: "conventional commits format"
  - synced: "merged/rebased with target branch"

# === NEVER COMMIT WHEN ===

  - "Build has errors"
  - "Tests failing"
  - "Feature incomplete (would break functionality)"
  - "Contains WIP/TODO in critical logic"
  - "Contains debugging code (console.log, print)"
  - "Contains commented-out code blocks"

# === COMMIT TIMING ===

  - "Completed functional unit"
  - "Specific bug fixed with regression test"
  - "Independent refactor (all tests pass)"
  - "Runnable state"

bad_times:
  - "Build failures"
  - "Test failures"
  - "Incomplete features"
  - "Experimental code with TODOs"

# === GRANULARITY ===
ideal_size:
  files: "1-10 (split if >10)"
  lines: "50-300"
  scope: "single concern"

split_principle:
  combine: ["feature + its tests", "tightly related multi-file"]
  separate: ["Feature A + Feature B", "refactor + new feature", "bugfix + incidental refactor"]
